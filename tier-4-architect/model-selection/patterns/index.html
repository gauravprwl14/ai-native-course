Model Selection Patterns

Pattern 1: Model Cascade​

Try the cheapest model first. Escalate to a more capable model only if confidence is below a threshold.

import anthropic

def cascade_call(prompt: str, confidence_threshold: float = 0.8) -> str:
    """
    Try Haiku first. If response confidence is low, escalate to Sonnet.
    Uses structured output to extract confidence.
    """
    client = anthropic.Anthropic()

    # Stage 1: Fast, cheap model
    haiku_response = client.messages.create(
        model="claude-3-haiku-20240307",
        max_tokens=512,
        system="""Answer the question. Rate your confidence from 0.0 to 1.0.
Respond as JSON: {"answer": "...", "confidence": 0.95}""",
        messages=[{"role": "user", "content": prompt}]
    )

    import json
    try:
        result = json.loads(haiku_response.content[0].text)
        confidence = result.get("confidence", 0.0)

        if confidence >= confidence_threshold:
            print(f"Haiku answered with confidence {confidence:.2f} — no escalation needed")
            return result["answer"]
    except (json.JSONDecodeError, KeyError):
        pass  # fall through to escalation

    # Stage 2: Escalate to capable model
    print("Confidence too low — escalating to Sonnet")
    sonnet_response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=1024,
        messages=[{"role": "user", "content": prompt}]
    )
    return sonnet_response.content[0].text


# Example usage
answer = cascade_call("What is the capital of France?")  # Haiku handles this
answer = cascade_call("Explain the derivation of the Euler-Lagrange equation.")  # Escalates to Sonnet

When to use: High-volume services where most queries are simple. A cascade can cut costs by 60–80% while maintaining quality for complex queries.

Gotcha: The confidence extraction relies on the model self-reporting accurately. Calibrate the threshold by testing on labeled examples where you know the correct answer.

Pattern 2: Task-Specific Routing​

Classify the incoming request first, then route to the model best suited for that task type.

from enum import Enum
import anthropic

class TaskType(Enum):
    SIMPLE_FACT = "simple_fact"       # Haiku
    CODE_GENERATION = "code"          # Sonnet
    COMPLEX_REASONING = "reasoning"   # Sonnet
    SUMMARIZATION = "summarize"       # Haiku
    UNKNOWN = "unknown"               # Sonnet (safe default)

MODEL_FOR_TASK = {
    TaskType.SIMPLE_FACT:      "claude-3-haiku-20240307",
    TaskType.SUMMARIZATION:    "claude-3-haiku-20240307",
    TaskType.CODE_GENERATION:  "claude-3-5-sonnet-20241022",
    TaskType.COMPLEX_REASONING:"claude-3-5-sonnet-20241022",
    TaskType.UNKNOWN:          "claude-3-5-sonnet-20241022",
}

def classify_task(prompt: str) -> TaskType:
    """Use a tiny model to classify the task type."""
    client = anthropic.Anthropic()
    response = client.messages.create(
        model="claude-3-haiku-20240307",
        max_tokens=10,
        system=f"""Classify this request as ONE of: {[t.value for t in TaskType]}.
Respond with ONLY the category name.""",
        messages=[{"role": "user", "content": prompt[:500]}]  # truncate for speed
    )
    label = response.content[0].text.strip().lower()
    for task in TaskType:
        if task.value == label:
            return task
    return TaskType.UNKNOWN


def route_and_call(prompt: str) -> tuple[str, str]:
    """Route to the best model and return (response, model_used)."""
    task_type = classify_task(prompt)
    model = MODEL_FOR_TASK[task_type]
    print(f"Task classified as: {task_type.value} → using {model}")

    client = anthropic.Anthropic()
    response = client.messages.create(
        model=model,
        max_tokens=1024,
        messages=[{"role": "user", "content": prompt}]
    )
    return response.content[0].text, model

When to use: When you have clearly distinct task types with different capability requirements. Works best when task categories are well-defined and the classifier is reliable.

Pattern 3: A/B Testing Models in Production (Shadow Mode)​

Test a new model against the current production model without impacting users.

import threading
import time
from dataclasses import dataclass

@dataclass
class ABResult:
    control_response: str
    control_latency_ms: float
    shadow_response: str
    shadow_latency_ms: float
    prompt: str

def shadow_test(
    prompt: str,
    control_model: str = "claude-3-haiku-20240307",
    shadow_model: str = "claude-3-5-sonnet-20241022",
) -> ABResult:
    """
    Run both models in parallel. Return control response to user immediately.
    Log shadow response for offline comparison.
    """
    import anthropic
    client = anthropic.Anthropic()

    shadow_result = {}

    def run_shadow():
        start = time.time()
        resp = client.messages.create(
            model=shadow_model,
            max_tokens=512,
            messages=[{"role": "user", "content": prompt}]
        )
        shadow_result["response"] = resp.content[0].text
        shadow_result["latency_ms"] = (time.time() - start) * 1000

    # Start shadow call in background
    shadow_thread = threading.Thread(target=run_shadow, daemon=True)
    shadow_thread.start()

    # Run control model — this is what the user sees
    start = time.time()
    control_resp = client.messages.create(
        model=control_model,
        max_tokens=512,
        messages=[{"role": "user", "content": prompt}]
    )
    control_latency = (time.time() - start) * 1000

    # Wait for shadow to finish (don't block user)
    shadow_thread.join(timeout=30)

    return ABResult(
        control_response=control_resp.content[0].text,
        control_latency_ms=control_latency,
        shadow_response=shadow_result.get("response", ""),
        shadow_latency_ms=shadow_result.get("latency_ms", 0),
        prompt=prompt,
    )

When to use: Before migrating to a new model. Collect thousands of shadow comparisons, have humans rate them, and only switch if the shadow model wins.

Pattern 4: Benchmark Harness​

Test the same set of prompts across multiple models. Compare accuracy, latency, and cost.

import time
from dataclasses import dataclass

@dataclass
class TestCase:
    prompt: str
    expected_keywords: list[str]  # response must contain these words

@dataclass
class BenchmarkResult:
    model: str
    test_case_id: int
    response: str
    passed: bool
    latency_ms: float
    input_tokens: int
    output_tokens: int

def evaluate_response(response: str, expected_keywords: list[str]) -> bool:
    """Check if response contains all expected keywords (case insensitive)."""
    return all(kw.lower() in response.lower() for kw in expected_keywords)

def run_benchmark(
    test_cases: list[TestCase],
    models: list[str],
) -> list[BenchmarkResult]:
    import anthropic
    client = anthropic.Anthropic()
    results = []

    for model in models:
        for i, tc in enumerate(test_cases):
            start = time.time()
            resp = client.messages.create(
                model=model,
                max_tokens=512,
                messages=[{"role": "user", "content": tc.prompt}]
            )
            latency_ms = (time.time() - start) * 1000
            response_text = resp.content[0].text

            results.append(BenchmarkResult(
                model=model,
                test_case_id=i,
                response=response_text,
                passed=evaluate_response(response_text, tc.expected_keywords),
                latency_ms=latency_ms,
                input_tokens=resp.usage.input_tokens,
                output_tokens=resp.usage.output_tokens,
            ))

    return results

def print_benchmark_summary(results: list[BenchmarkResult]):
    models = sorted(set(r.model for r in results))
    print(f"\n{'Model':<35} {'Accuracy':>10} {'Avg Latency':>12} {'Avg Input Tokens':>18}")
    print("-" * 80)
    for model in models:
        model_results = [r for r in results if r.model == model]
        accuracy = sum(r.passed for r in model_results) / len(model_results) * 100
        avg_latency = sum(r.latency_ms for r in model_results) / len(model_results)
        avg_tokens = sum(r.input_tokens for r in model_results) / len(model_results)
        print(f"{model:<35} {accuracy:>9.1f}% {avg_latency:>11.0f}ms {avg_tokens:>17.0f}")

Anti-Patterns​

Anti-PatternWhat Goes WrongFix
Always using the biggest model10–50x cost for tasks a small model handles perfectlyRun a benchmark harness on your own test cases first
Not benchmarking your specific taskMMLU says Model A is better, but Model B does better on your RAG queriesBuild a 50–100 case test set from your actual data
Single-model architectureNo ability to trade off cost vs quality per requestDesign for model substitutability from the start
Ignoring latency in benchmarksYou pick the most accurate model and discover it's 3x slowerAlways measure latency alongside accuracy
