# Patterns: MCP Server Patterns

---

## Pattern 1: Filesystem MCP Server

Expose a directory as both resources (for reading files) and tools (for listing/searching). This is the most common MCP server pattern — it lets Claude Desktop read your codebase.

```python
import os
from dataclasses import dataclass, field
from typing import Any

@dataclass
class MCPTool:
    name: str
    description: str
    input_schema: dict
    handler: Any = field(default=None, repr=False)

@dataclass
class MCPResource:
    uri: str
    name: str
    description: str
    mime_type: str = "text/plain"

class MCPServer:
    def __init__(self, name: str):
        self.name = name
        self._tools: dict[str, MCPTool] = {}
        self._resources: dict[str, MCPResource] = {}

    def register_tool(self, tool: MCPTool) -> None:
        self._tools[tool.name] = tool

    def register_resource(self, resource: MCPResource) -> None:
        self._resources[resource.uri] = resource

    def call_tool(self, name: str, arguments: dict) -> dict:
        if name not in self._tools:
            raise ValueError(f"Unknown tool: {name}")
        try:
            result = self._tools[name].handler(arguments)
            return {"content": result, "isError": False}
        except Exception as e:
            return {"content": str(e), "isError": True}


def create_filesystem_server(base_dir: str) -> MCPServer:
    server = MCPServer("filesystem")

    def list_files_handler(args: dict) -> list[str]:
        return sorted(os.listdir(base_dir))

    def read_file_handler(args: dict) -> str:
        filename = args["filename"]
        # Security: prevent path traversal
        full_path = os.path.realpath(os.path.join(base_dir, filename))
        if not full_path.startswith(os.path.realpath(base_dir)):
            raise ValueError("Access denied: path outside base directory")
        with open(full_path) as f:
            return f.read()

    server.register_tool(MCPTool(
        name="list_files",
        description="List files in the project directory",
        input_schema={"type": "object", "properties": {}},
        handler=list_files_handler,
    ))

    server.register_tool(MCPTool(
        name="read_file",
        description="Read the contents of a file",
        input_schema={
            "type": "object",
            "properties": {"filename": {"type": "string"}},
            "required": ["filename"],
        },
        handler=read_file_handler,
    ))

    return server
```

**When to use:** Letting an AI host read project files, configs, logs, or documentation. Common in developer tools (Cursor, VS Code extensions).

**Security note:** Always use `os.path.realpath` to resolve symlinks before checking the path prefix. Naive `startswith` checks can be defeated with `../` traversal.

## Pattern 2: Database MCP Server

Wrap a database as a set of query tools. The model can explore the schema, run read-only queries, and retrieve specific records.

```python
import sqlite3

def create_database_server(db_path: str) -> MCPServer:
    server = MCPServer("database")

    def list_tables_handler(args: dict) -> list[str]:
        with sqlite3.connect(db_path) as conn:
            cursor = conn.execute(
                "SELECT name FROM sqlite_master WHERE type='table'"
            )
            return [row[0] for row in cursor.fetchall()]

    def describe_table_handler(args: dict) -> list[dict]:
        table = args["table_name"]
        with sqlite3.connect(db_path) as conn:
            cursor = conn.execute(f"PRAGMA table_info({table})")
            columns = cursor.fetchall()
        return [
            {"name": col[1], "type": col[2], "nullable": not col[3]}
            for col in columns
        ]

    def query_handler(args: dict) -> list[dict]:
        sql = args["sql"]
        # Guard: only allow SELECT statements
        if not sql.strip().upper().startswith("SELECT"):
            raise ValueError("Only SELECT queries are allowed")
        with sqlite3.connect(db_path) as conn:
            conn.row_factory = sqlite3.Row
            cursor = conn.execute(sql)
            return [dict(row) for row in cursor.fetchall()]

    server.register_tool(MCPTool(
        name="list_tables",
        description="List all tables in the database",
        input_schema={"type": "object", "properties": {}},
        handler=list_tables_handler,
    ))

    server.register_tool(MCPTool(
        name="describe_table",
        description="Get the schema of a specific table",
        input_schema={
            "type": "object",
            "properties": {"table_name": {"type": "string"}},
            "required": ["table_name"],
        },
        handler=describe_table_handler,
    ))

    server.register_tool(MCPTool(
        name="query",
        description="Run a read-only SQL SELECT query",
        input_schema={
            "type": "object",
            "properties": {
                "sql": {"type": "string", "description": "SQL SELECT statement"},
            },
            "required": ["sql"],
        },
        handler=query_handler,
    ))

    return server
```

**When to use:** Letting the model explore and query a database for analytics, reporting, or data exploration tasks.

**Security note:** Always restrict to SELECT statements. Consider using a read-only database connection or a read-only user with appropriate DB permissions.

## Pattern 3: API Wrapper MCP Server

Wrap a REST API as MCP tools. The model can call weather APIs, internal services, or any HTTP endpoint through a clean tool interface.

```python
import urllib.request
import urllib.parse
import json

def create_weather_server(api_base_url: str, api_key: str) -> MCPServer:
    server = MCPServer("weather")

    def get_current_weather_handler(args: dict) -> dict:
        city = args["city"]
        params = urllib.parse.urlencode({"q": city, "appid": api_key, "units": "metric"})
        url = f"{api_base_url}/weather?{params}"
        with urllib.request.urlopen(url) as response:
            data = json.loads(response.read())
        return {
            "city": data["name"],
            "temperature_c": data["main"]["temp"],
            "description": data["weather"][0]["description"],
            "humidity_pct": data["main"]["humidity"],
        }

    def get_forecast_handler(args: dict) -> list[dict]:
        city = args["city"]
        days = min(args.get("days", 3), 5)  # cap at 5 days
        params = urllib.parse.urlencode({
            "q": city, "appid": api_key, "units": "metric", "cnt": days * 8
        })
        url = f"{api_base_url}/forecast?{params}"
        with urllib.request.urlopen(url) as response:
            data = json.loads(response.read())
        # Return one entry per day (every 8th 3-hour slot)
        return [
            {
                "date": item["dt_txt"][:10],
                "temp_c": item["main"]["temp"],
                "description": item["weather"][0]["description"],
            }
            for item in data["list"][::8][:days]
        ]

    server.register_tool(MCPTool(
        name="get_current_weather",
        description="Get the current weather for a city",
        input_schema={
            "type": "object",
            "properties": {"city": {"type": "string"}},
            "required": ["city"],
        },
        handler=get_current_weather_handler,
    ))

    server.register_tool(MCPTool(
        name="get_forecast",
        description="Get the weather forecast for a city",
        input_schema={
            "type": "object",
            "properties": {
                "city": {"type": "string"},
                "days": {"type": "integer", "minimum": 1, "maximum": 5},
            },
            "required": ["city"],
        },
        handler=get_forecast_handler,
    ))

    return server
```

**When to use:** Exposing any third-party API (weather, maps, ticketing, internal microservices) as reusable MCP tools.

## Pattern 4: Multi-Server Composition

A host can connect to multiple MCP servers simultaneously. Tools and resources from all servers are pooled for the model. Here's how you might use a composed setup in a custom host:

```python
class ComposedMCPHost:
    """A simple in-process host that composes multiple MCP servers."""

    def __init__(self):
        self._servers: dict[str, MCPServer] = {}

    def add_server(self, server: MCPServer) -> None:
        self._servers[server.name] = server

    def list_all_tools(self) -> list[dict]:
        """Return all tools from all servers, namespaced by server name."""
        tools = []
        for server_name, server in self._servers.items():
            for tool in server.list_tools():
                # Namespace the tool to avoid name collisions
                tools.append({
                    **tool,
                    "name": f"{server_name}__{tool['name']}",
                    "_server": server_name,
                })
        return tools

    def call_tool(self, namespaced_name: str, arguments: dict) -> dict:
        """Route a tool call to the correct server."""
        if "__" not in namespaced_name:
            raise ValueError(f"Tool name must be namespaced: {namespaced_name}")
        server_name, tool_name = namespaced_name.split("__", 1)
        if server_name not in self._servers:
            raise ValueError(f"Unknown server: {server_name}")
        return self._servers[server_name].call_tool(tool_name, arguments)


# Usage
host = ComposedMCPHost()
host.add_server(create_filesystem_server("/my/project"))
host.add_server(create_database_server("./data.db"))

all_tools = host.list_all_tools()
# → [{"name": "filesystem__list_files", ...}, {"name": "database__query", ...}, ...]

result = host.call_tool("filesystem__list_files", {})
```

**When to use:** Production applications where the model needs access to multiple independent data sources simultaneously.

## Anti-Patterns

| Anti-pattern | What Goes Wrong | Fix |
|--------------|----------------|-----|
| **Business logic in MCP transport** | Tool handlers contain auth checks, caching, retry logic, data transformation. They become unmaintainable blobs. | Keep handlers thin: validate input, call a service layer, return result. Business logic lives in the service layer. |
| **Not validating tool inputs** | `read_file({"filename": "../../etc/passwd"})` reads outside the intended directory | Validate and sanitise all inputs. Use `os.path.realpath` + prefix check for path parameters. |
| **Leaking secrets in resource content** | A resource returns a `.env` file or a config that includes API keys | Never expose sensitive files as resources. Filter sensitive fields programmatically. |
| **Mutable server state without locks** | Multiple concurrent tool calls race on shared state (cache, counter, connection pool) | Use thread-safe data structures or restrict each server to a single thread per connection |
| **Not returning `isError: true` on failure** | Tool raises an exception, server crashes, host loses the connection | Wrap all tool handlers in try/except; return structured error responses |
